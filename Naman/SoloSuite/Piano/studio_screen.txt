<script>
    // Variables to track time
    const startLoadTime = Date.now();
    const minimumTime = 2000; // 2 Seconds minimum

    /**
     * Call this function whenever your page is "ready"
     * (e.g., inside an 'onload' or when your audio finishes loading)
     */
    function finishLoading() {
        const timePassed = Date.now() - startLoadTime;
        
        if (timePassed < minimumTime) {
            // Wait for the remainder of the 2 seconds
            setTimeout(hideScreen, minimumTime - timePassed);
        } else {
            hideScreen();
        }
    }

    /**
     * Manual bypass if the user clicks your name
     */
    function bypassLoading() {
        hideScreen();
    }

    function hideScreen() {
        const overlay = document.getElementById('loading-overlay');
        overlay.classList.add('hidden');
    }

    // Example Use Case: 
    // If you don't have a real asset loader, you can test it by 
    // running finishLoading() after the window loads:

</script>

    <script>
        // --- LOADING LOGIC (Integrated from your request) ---
        const startLoadTime = Date.now();
        const minimumTime = 2000; // 2 Seconds

        function finishLoading() {
            const timePassed = Date.now() - startLoadTime;
            if (timePassed < minimumTime) {
                setTimeout(hideScreen, minimumTime - timePassed);
            } else {
                hideScreen();
            }
        }

        function bypassLoading() { hideScreen(); }
        function hideScreen() { document.getElementById('loading-overlay').classList.add('hidden'); }

        // --- ASSET LOADER ---
        const NOTES = ['C', 'C#', 'D', 'D#', 'E', 'F', 'F#', 'G', 'G#', 'A', 'A#', 'B'];
        const buffers = new Tone.ToneAudioBuffers();
        const activeSources = new Map();

        const reverb = new Tone.Reverb(2.5).toDestination();
        const limiter = new Tone.Limiter(-1).connect(reverb);

        function init() {
            let idx = NOTES.indexOf('A'), oct = 0;
            for (let i = 0; i < 88; i++) {
                const n = NOTES[idx] + oct;
                // Add your asset loading here
                buffers.add(n, `./assets/${n}.mp3`, () => {
                    if (buffers.loaded) finishLoading(); // Trigger our loading check
                });
                if (idx === 11) { idx = 0; oct++; } else { idx++; }
            }
            renderPiano();
        }

        function playNote(note) {
            if (!buffers.has(note)) return;
            Tone.start();
            const source = new Tone.ToneBufferSource(buffers.get(note));
            source.loop = true; source.loopStart = 0.5; source.loopEnd = 2.5;
            source.connect(limiter);
            source.start();
            activeSources.set(note, source);
            document.getElementById(`key-${note}`)?.classList.add('pressed');
        }

        function stopNote(note) {
            const s = activeSources.get(note);
            if (s) { s.loop = false; s.fadeOut = 2; s.stop("+2.1"); activeSources.delete(note); }
            document.getElementById(`key-${note}`)?.classList.remove('pressed');
        }

        function renderPiano() {
            const container = document.getElementById('piano-keys');
            let idx = NOTES.indexOf('A'), oct = 0;
            for (let i = 0; i < 88; i++) {
                const n = NOTES[idx] + oct;
                const div = document.createElement('div');
                div.className = `key ${n.includes('#') ? 'black' : 'white'}`;
                div.id = `key-${n}`;
                div.onmousedown = () => playNote(n);
                div.onmouseup = () => stopNote(n);
                container.appendChild(div);
                if (idx === 11) { idx = 0; oct++; } else { idx++; }
            }
        }

        // --- THE "EXAMPLE USE CASE" SAFETY NET ---
        // If the samples load instantly or if there is no loader, 
        // this ensures the branding screen stays for the 2s minimum.
        window.onload = finishLoading;

        init();
    </script>
